```javascript
    p.strokeWeight(2);
    p.line(0, 0, x, y);
```

Example 3: "Solar System" - NO controls, fully automatic:
```javascript
const visualDemo = document.getElementById('visual-demo');
visualDemo.innerHTML = '<div id="solar-canvas"></div>';

let orbitalSpeed = 1.0;
let angles = [0, 0, 0, 0];

const sketch = (p) => {{
  p.setup = () => {{
    const canvasWidth = Math.min(850, visualDemo.clientWidth - 40);
    const canvas = p.createCanvas(canvasWidth, 400);
    canvas.parent('solar-canvas');
  }};
  p.draw = () => {{
    p.background(10, 20, 40);
    p.translate(p.width/2, 200);
    
    // Sun
    p.fill(255, 200, 50);
    p.noStroke();
    p.circle(0, 0, 50);
    
    // Planets with orbits
    const colors = [[100,150,255], [255,100,100], [100,255,150], [255,200,100]];
    for(let i = 0; i < 4; i++) {{
      const radius = 80 + i * 45;
      const size = 15 + i * 3;
      
      // Orbit
      p.noFill();
      p.stroke(60);
      p.strokeWeight(1);
      p.circle(0, 0, radius * 2);
      
      // Planet
      const x = p.cos(angles[i]) * radius;
      const y = p.sin(angles[i]) * radius;
      p.fill(...colors[i]);
      p.noStroke();
      p.circle(x, y, size);
      
      angles[i] += (0.01 / (i + 1)) * orbitalSpeed;
    }}
  }};
}};
const myP5 = new p5(sketch);
if (!window.p5Instances) window.p5Instances = [];
window.p5Instances.push(myP5);
```

Example 4: "Parabola" - NO controls needed:
```javascript
const visualDemo = document.getElementById('visual-demo');
visualDemo.innerHTML = '<div id="parabola-canvas"></div>';

const sketch = (p) => {{
  p.setup = () => {{
    const canvasWidth = Math.min(850, visualDemo.clientWidth - 40);
    const canvas = p.createCanvas(canvasWidth, 400);
    canvas.parent('parabola-canvas');
  }};
  
  p.draw = () => {{
    p.background(20, 30, 50);
    
    // Grid
    p.stroke(40, 50, 70);
    p.strokeWeight(1);
    for(let i = 0; i < p.width; i += 50) p.line(i, 0, i, p.height);
    for(let i = 0; i < p.height; i += 50) p.line(0, i, p.width, i);
    
    // Parabola
    p.stroke(0, 255, 150);
    p.strokeWeight(3);
    p.noFill();
    p.beginShape();
    for(let x = 0; x < p.width; x++) {{
      let normX = (x - p.width/2) / 100;
      let y = p.height/2 - (normX * normX * 50);
      p.vertex(x, y);
    }}
    p.endShape();
  }};
}};
const myP5 = new p5(sketch);
if (!window.p5Instances) window.p5Instances = [];
```

```javascript
let speed = 2.0;
let animation;

function updateAnimation() {{
  const container = document.getElementById('car-container');
  const maxX = container ? container.offsetWidth - 100 : 750;
  if(animation) animation.kill();
  animation = gsap.to('.car', {{
    duration: speed, 
    x: maxX, 
    repeat: -1, 
    yoyo: true, 
    ease: 'power1.inOut'
  }});
}}

document.getElementById('speedSlider').addEventListener('input', (e) => {{
  speed = parseFloat(e.target.value);
  document.getElementById('speedVal').textContent = speed.toFixed(1);
  updateAnimation();
}});

updateAnimation();
```

Example 7: "Canvas Animation" - NO controls, fully automatic:
```javascript
const visualDemo = document.getElementById('visual-demo');
visualDemo.innerHTML = '<div id="solar-canvas"></div>';

const canvas = document.createElement('canvas');
const canvasWidth = Math.min(850, visualDemo.clientWidth - 40);
canvas.width = canvasWidth;
canvas.height = 400;
canvas.style.display = 'block';
canvas.style.margin = '0 auto';
canvas.style.maxWidth = '100%';
document.getElementById('solar-canvas').appendChild(canvas);
const ctx = canvas.getContext('2d');

let angle = 0;
function animate() {{
  ctx.fillStyle = 'rgba(0, 0, 0, 0.1)';
  ctx.fillRect(0, 0, canvas.width, canvas.height);
  
  // Sun
  ctx.fillStyle = '#FDB813';
  ctx.beginPath();
  ctx.arc(canvas.width/2, 200, 30, 0, Math.PI * 2);
  ctx.fill();
  
  // Planet
  const x = canvas.width/2 + Math.cos(angle) * 150;
  const y = 200 + Math.sin(angle) * 150;
  ctx.fillStyle = '#4169E1';
  ctx.beginPath();
  ctx.arc(x, y, 15, 0, Math.PI * 2);
  ctx.fill();
  
  angle += 0.02;
  window.animationFrameId = requestAnimationFrame(animate);
}}
animate();
```
